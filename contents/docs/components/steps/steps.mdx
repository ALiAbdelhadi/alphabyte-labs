## Installation
<div className="not-prose">
<Step>
<StepItem title="Install Dependencies">
First , you need to install dependencies:
<Pre className="language-bash">
{`npm i clsx tailwind-merge`}
</Pre>
</StepItem>
  <StepItem title="Create utils.ts">
  Second, you need to create a new folder called <MdxBadge>lib</MdxBadge> in you app directory and create a new file called <MdxBadge>utils.ts</MdxBadge> and paste the following code in it: 
<Pre className="language-typescript" folderPath="lib/utils.ts" >
  {`import { type ClassValue, clsx } from "clsx";
import { twMerge } from "tailwind-merge";

export function cn(...inputs: ClassValue[]) {
    return twMerge(clsx(inputs));
}
`}
</Pre>
   </StepItem>
  <StepItem title="Create Step component">
    Now you can create in your <MdxBadge>components</MdxBadge> a Step.tsx file and add the following code in it :
    <CodeBlockWrapper>
    <Pre className="language-typescript" folderPath="components/Step.tsx">
    {`
    import { cn } from "@/lib/utils"
import clsx from "clsx"
import { Children, PropsWithChildren, ReactNode } from "react"

interface StepItemProps {
  title?: string
  children: ReactNode
}
interface StepProps {
  children: ReactNode
}

export function Step({ children }: PropsWithChildren<StepProps>) {
  const length = Children.count(children)
  return (
    <div className="flex flex-col my-5">
      {Children.map(children, (child, index) => (
        <div
          className={cn(
            "relative border-l pl-9",
            clsx({ "pb-5": index < length - 1 })
          )}
        >
          <div className="absolute -left-4 flex items-center justify-center w-8 h-8 rounded-full bg-muted text-sm font-code font-medium -indent-[1px]">
            {index + 1}
          </div>
          {child}
        </div>
      ))}
    </div>
  )
}

export function StepItem({ children, title }: StepItemProps) {
  return (
    <div className="space-y-4">
      {title && <h3 className="!mt-0 text-lg md:text-xl text-gray-950">{title}</h3>}
      <div className="text-base text-muted-foreground space-y-3">{children}</div>
    </div>
  )
}
    `}
    </Pre>
    </CodeBlockWrapper>
  </StepItem>
  <StepItem title="Be comfortable to use it and make it yours"/>
</Step>
    </div>

## Usage

<div className="not-prose space-y-6">
<Pre>{`import { Button } from "@/components/ui/button"
import {
  Popover,
  PopoverContent,
  PopoverTrigger,
} from "@/components/ui/popover"

const PopoverDemo = () => {
  return (
    <Popover>
      <PopoverTrigger asChild>
        <Button>Click me</Button>
      </PopoverTrigger>
      <PopoverContent>
        <p>Your content here</p>
      </PopoverContent>
    </Popover>
  )
}

export default PopoverDemo`}</Pre>
</div>

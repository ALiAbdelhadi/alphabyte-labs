## Usage
<div className="space-y-6">
  <Pre>
    {`import { Tooltip, TooltipTrigger, TooltipContent, TooltipProvider } from "@/components/ui/tooltip"`}
  </Pre>
  <Pre>
    {`<TooltipProvider>
    <Tooltip>
        <TooltipTrigger asChild>Hover</TooltipTrigger>
            <TooltipContent >
                <p>Default Tooltip</p>
            </TooltipContent>
        </Tooltip>
    </TooltipProvider>
`}
  </Pre>
</div>
